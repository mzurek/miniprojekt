Biomedical engineering - miniproject - data exploration¶

Projekt wykorzystuje baz danych Global Terrorism Database zawierajc informacje na temat atakw terrorystycznych na przestrzeni lat 1970-2017. Baza danych jest tworzona przez naukowcw z National Consortium for Study of Terrorism and Responses to Terrorism (START) - University of Maryland. Celem projektu byo opracowanie 5 pyta dotyczcych danych w bazie oraz zaprezentowanie odpowiedzi np. w postaci graficznej za pomoc wykresw.

def getSeason(day,month):
    if month in [4,5] or (month==3 and day>=21) or (month==6 and day<21):
        return 'spring'
    elif month in [7,8] or (month==6 and day>=21) or (month==9 and day<23):
        return 'summer'
    elif month in [10,11] or (month==9 and day>=23) or (month==12 and day<22):
        return 'autumn'
    else:
        return 'winter'
        


indexes = list(set(terror['Month'].nonzero()[0]).intersection(terror['Day'].nonzero()[0]))
data = terror.iloc[indexes,[1,2]]

for i in indexes:
    data.at[i,'Season'] = getSeason(data.at[i,'Day'],data.at[i,'Month'])
    

values = [sum(data['Season']=='spring'),sum(data['Season']=='summer'),sum(data['Season']=='autumn'),
         sum(data['Season']=='winter')]
labels = ["spring","summer","autumn","winter"]

for i in range(0,4):
    print(labels[i],": ",values[i])

step = 5
i = 0

for year in terror['Year']:
    is_not_between = True
    start_year = 1970
    end_year = 1974
    while is_not_between:
        if(year<start_year or year>end_year):
            is_not_between = True
            start_year += step
            end_year += step
        else:
            is_not_between = False
            if end_year>2017: end_year = 2017
            terror.at[i,'TimePeriod'] = str(start_year)+"-"+str(end_year)
            i += 1

pd.crosstab(terror.TimePeriod,terror.AttackType).plot.barh(stacked=True,width=1,color=sns.color_palette('RdYlGn',9))
